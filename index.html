import cv2
import numpy as np
import matplotlib.pyplot as plt

def detect_yellow(image_path):

    img = cv2.imread(image_path)


    hsv_img = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)


    yellow_lower = np.array([20, 100, 100])
    yellow_upper = np.array([30, 255, 255])


    mask = cv2.inRange(hsv_img, yellow_lower, yellow_upper)


    contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_NONE)

    output_img = img.copy()


    for contour in contours:
        x, y, w, h = cv2.boundingRect(contour)
        if cv2.contourArea(contour) > 500:
            cv2.rectangle(output_img, (x, y), (x + w, y + h), (0, 255, 0), 3)


    output_img_rgb = cv2.cvtColor(output_img, cv2.COLOR_BGR2RGB)


    plt.figure(figsize=(12, 6))


    plt.subplot(1, 2, 1)
    plt.imshow(cv2.cvtColor(img, cv2.COLOR_BGR2RGB))
    plt.title("Original Image")
    plt.axis("off")


    plt.subplot(1, 2, 2)
    plt.imshow(output_img_rgb)
    plt.title("Yellow Color Detection")
    plt.axis("off")

    plt.show()

    return output_img

import cv2, numpy as np

cap = cv2.VideoCapture(0)
N = 800

ret, frame = cap.read()
x, y, w, h = cv2.selectROI(frame,False)

roi = frame[y:y+h, x:x+w]
ref_color = roi.mean(axis=(0,1))

particles = np.tile([x+w/2, y+h/2], (N,1))
weights = np.ones(N) / N

while True:
    ret, frame = cap.read()
    if not ret: break

    h_img, w_img = frame.shape[:2]


    particles += np.random.randn(N,2) * 15


    particles[:,0] = np.clip(particles[:,0], 0, w_img-1)
    particles[:,1] = np.clip(particles[:,1], 0, h_img-1)


    coords = particles.astype(int)
    colors = frame[coords[:,1], coords[:,0]]
    dist = np.linalg.norm(colors - ref_color, axis=1)
    weights = 1 / (dist + 1)  
    weights /= weights.sum()


    pos = np.average(particles, weights=weights, axis=0).astype(int)


    idx = np.random.choice(N, N, p=weights)
    particles = particles[idx]
    weights.fill(1.0 / N)


    for p in particles:
      cv2.circle(frame, tuple(p.astype(int)), 1, (0,255,0), -1)
    cv2.circle(frame, tuple(pos), 8, (0,0,255), -1)

    cv2.imshow("Particle Filter", frame)
    if cv2.waitKey(30) & 0xFF == 27 :
       break

cap.release()
cv2.destroyAllWindows()

import cv2
import mediapipe as mp
from mediapipe.tasks.python import vision
from mediapipe.tasks.python.core.base_options import BaseOptions


model_path = "hand_landmarker.task"


options = vision.HandLandmarkerOptions(
    base_options=BaseOptions(model_asset_path=model_path),
    num_hands=2
)
hand_landmarker = vision.HandLandmarker.create_from_options(options)


def recognize_number(hand_landmarks):
    fingers = []

    lm = hand_landmarks

    if lm[4].x < lm[3].x:
        fingers.append(1)
    else:
        fingers.append(0)


    tips = [8, 12, 16, 20]
    for tip in tips:
        if lm[tip].y < lm[tip - 2].y:
            fingers.append(1)
        else:
            fingers.append(0) 
            
    total_fingers = sum(fingers)


    if fingers == [0,0,0,0,0]:
        return "0"
    elif fingers == [0,1,0,0,0]:
        return "1"
    elif fingers == [0,1,1,0,0]:
        return "2"
    elif fingers == [0,1,1,1,0]:
        return "3"
    elif fingers == [0,1,1,1,1]:
        return "4"
    elif fingers == [1,1,1,1,1]:
        return "5"
    elif fingers == [1,0,0,0,1]:
        return "6"
    elif fingers == [1,1,0,0,1]:
        return "7"
    elif fingers == [1,1,1,0,1]:
        return "8"
    elif fingers == [1,1,1,1,0]:
        return "9"
    else:
        return str(total_fingers)


cap = cv2.VideoCapture(0)

while True:
    ret, frame = cap.read()
    frame = cv2.flip(frame, 1) # flipping horizontaly
    if not ret:
        break


    rgb_frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)


    mp_image = mp.Image(image_format=mp.ImageFormat.SRGB, data=rgb_frame)


    result = hand_landmarker.detect(mp_image)

    if result.hand_landmarks:
        for hand_landmarks in result.hand_landmarks:
            gesture = recognize_number(hand_landmarks)


            for lm in hand_landmarks:
                h, w, _ = frame.shape
                cx, cy = int(lm.x * w), int(lm.y * h)
                cv2.circle(frame, (cx, cy), 5, (0, 255, 0), -1)


            h, w, _ = frame.shape
            x = int(hand_landmarks[0].x * w)
            y = int(hand_landmarks[0].y * h)
            
            cv2.putText(frame, gesture, (x, y - 20), 
                        cv2.FONT_HERSHEY_SIMPLEX, 1, 
                        (0, 0, 0), 2)

    cv2.imshow("Hand Gesture Recognition", frame)

    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()

import cv2
import numpy as np

image_path = 'object.jpg'
image = cv2.imread(image_path)

if image is None:
    print(f"Error: Could not load image at {image_path}. Check the file name and path.")
else:

    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    

    blur = cv2.GaussianBlur(gray, (13, 13), 0)
    

    thresh = cv2.adaptiveThreshold(
        blur, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
        cv2.THRESH_BINARY_INV, 11, 2
    )


    kernel = np.ones((5, 5), np.uint8)

    closing = cv2.morphologyEx(thresh, cv2.MORPH_CLOSE, kernel, iterations=3)

    opening = cv2.morphologyEx(closing, cv2.MORPH_OPEN, kernel, iterations=2)


    contours, _ = cv2.findContours(opening, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

    count = 0
    for cnt in contours:
        area = cv2.contourArea(cnt)
        

        if area > 1500:
            count += 1
            cv2.drawContours(image, [cnt], -1, (0, 0, 255), 3)


    print(f"Number of Objects: {count}")


    cv2.imshow("Threshold (Cleaned)", opening)
    cv2.imshow("Result (Counted)", image)


    cv2.waitKey(0)
    cv2.destroyAllWindows()
