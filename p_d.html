%{
#include 
#include 
%}

%token NUMBER

%%

input:
      input line
    | 
    ;

line:
      '\n'
    | expr '\n' { printf("Result = %d\n", $1); }
    ;

expr:
      expr '+' expr   { $$ = $1 + $3; }
    | expr '-' expr   { $$ = $1 - $3; }
    | expr '*' expr   { $$ = $1 * $3; }
    | expr '/' expr   { $$ = $1 / $3; }
    | '(' expr ')'    { $$ = $2; }
    | NUMBER          { $$ = $1; }
    ;

%%

int main() {
    printf("Enter Expression:\n");
    yyparse();
    return 0;
}

int yyerror(char *s) {
    printf("Error: %s\n", s);
    return 0;
}

Count vowels and Cosanants


%{
#include 

int vowels = 0;
int consonants = 0;
%}

%%
[aAeEiIoOuU]     { vowels++; }

[b-df-hj-np-tv-zB-DF-HJ-NP-TV-Z]   { consonants++; }

[ \t\n]          ;   // ignore spaces

.                ;   // ignore other characters
%%

int main() {
    printf("Enter a string:\n");
    yylex();

    printf("Vowels = %d\n", vowels);
    printf("Consonants = %d\n", consonants);

    return 0;
}

Count no. of char,words,spaces and lines


%{
#include 

int chars = 0;
int words = 0;
int spaces = 0;
int lines = 0;
%}

%%
[ \t]        { spaces++; chars++; }

\n           { lines++; chars++; }

[a-zA-Z]+    { words++; chars += yyleng; }

.            { chars++; }
%%

int main() {
    printf("Enter text:\n");
    yylex();

    printf("\nCharacters = %d", chars);
    printf("\nWords = %d", words);
    printf("\nSpaces = %d", spaces);
    printf("\nLines = %d\n", lines);

    return 0;
}


Count positive and negative integer and fractions.



%{
#include 

int pos_int = 0, neg_int = 0;
int fraction = 0;
%}

%%
[-]?[0-9]+"."[0-9]+    { fraction++; }

[-][0-9]+              { neg_int++; }

[0-9]+                 { pos_int++; }

[ \t\n]                ;   // ignore spaces

.                      ;   // ignore others
%%

int main() {
    printf("Enter input:\n");
    yylex();

    printf("\nPositive Integers = %d", pos_int);
    printf("\nNegative Integers = %d", neg_int);
    printf("\nFractions = %d\n", fraction);

    return 0;
}
